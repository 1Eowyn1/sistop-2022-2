                public static int checkOption(String input)
                {
                    int parsed = -1;
                    try <-Un try-catch podría llegar a ser una llamada al sistema si no se atrapa adecuadamente la exceción
                    {
                        parsed = Int32.Parse(input);
                        if (parsed < 1 || parsed > 4) {
                            Console.WriteLine("Opcion no disponible");
                            return 5;
                        }
                    }
                    catch (FormatException)
                    {
                        Console.WriteLine("Entrada incorrecta");
                        return 5;
                    }
                    return parsed;
                }
        
                private static void calculateCube()
                {
                    int number = isInput();
                    Console.WriteLine("Su Cubo es: " + (Math.Pow(number, 3)));
                    Thread.Sleep(3000); <-El detener la ejecución del hilo del programa es una llamada al sistema, regresando el control al programa hasta que se cumpla la condición (3000 milisegundos de espera)
                    Console.Clear(); <-Como se limpia la pantalla en la que se está ejecutando el programa es necesario administrar la impresión de la imagen en pantalla
                }
        
                private static void showMatrix() 
                { 
                    for ( int i = 0; i < 3 ; i++)
                    {
                        for (int j = 1; j < 4; j++)
                        {
                            Console.Write("[" + (3 * i + j) + "] ","");
                        }
                        Console.WriteLine();
                    }
                    Thread.Sleep(3000);
                    Console.Clear();
                }
                 private static void reverseNumber()
                {
                    int number = isInput();
                    string str = number.ToString();
                    char[] charArray = str.ToCharArray();
                    Array.Reverse(charArray);
                    Console.WriteLine(new string(charArray));
                    Thread.Sleep(3000);
                    Console.Clear();
                }
        
                private static int isInput()
                {
                    int parsed = 0;
                    Console.WriteLine("Ingresa el numero:");
